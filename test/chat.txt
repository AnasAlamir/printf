int print_c(char a, int *i)
{
    printf("%c", a);
    (*i)++;
    return (1);
}

int print_s(char *a, int *i)
{
    if (a == NULL)
    {
        printf("(null)");
        (*i) += 6;
        return (0);
    }
    printf("%s", a);
    (*i) += strlen(a);
    return (1);
}
